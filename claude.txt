class MycelialNode {
  constructor(init=false){
    this.x = Math.random()*W;
    this.y = init ? Math.random()*H : H + 20;
    this.vx = (Math.random()-.5)*.18;
    this.vy = -(Math.random()*.22 + .04);
    this.size = Math.random()*1.8 + .4;
    this.alpha = 0;
    this.maxAlpha = Math.random()*.3 + .06;
    this.life = 0;
    this.maxLife = Math.random()*700 + 400;
    this.phase = Math.random()*Math.PI*2; // breath offset
    // Mouse-burst flag: nodes spawned from mouse presence glow brighter
    this.fromMouse = false;
    const r = Math.random();
    if(r < .55) this.hue = '126,200,80';       // glow-green
    else if(r < .82) this.hue = '200,160,64';   // gold
    else this.hue = '240,232,208';               // cream
  }
  update(t){
    // Drift toward/away from mouse with organic wavering
    const dx = mouseX - this.x, dy = mouseY - this.y;
    const dist = Math.sqrt(dx*dx+dy*dy);
    if(dist < MOUSE_INFLUENCE && dist > 0){
      const pull = (1 - dist/MOUSE_INFLUENCE) * (.0008 + mousePressure*.001);
      this.vx += dx * pull; this.vy += dy * pull;
      // Near mouse: brighten
      this.maxAlpha = Math.min(this.maxAlpha + .001, this.fromMouse ? .65 : .38);
    } else {
      this.maxAlpha = Math.max(this.maxAlpha - .0005, this.fromMouse ? .18 : .06);
    }
    // Dampen velocity (viscous medium)
    this.vx *= .985; this.vy *= .985;
    // Subtle wander
    this.vx += (Math.random()-.5)*.008;
    this.vy += (Math.random()-.5)*.004 - .015;

    this.x += this.vx; this.y += this.vy;
    this.life++;

    // Breathing alpha at resting-breath rate (sync with 10s body breath)
    const breathe = Math.sin(t*.00063 + this.phase) * .018;
    if(this.life < 80) this.alpha = this.maxAlpha*(this.life/80);
    else if(this.life > this.maxLife-100) this.alpha = this.maxAlpha*((this.maxLife-this.life)/100);
    else this.alpha = this.maxAlpha + breathe;

    if(this.life >= this.maxLife || this.y < -20) return true; // signal remove
    return false;
  }

    ctx.arc(this.x,this.y,this.size,0,Math.PI*2);
    ctx.fillStyle=`rgba(${this.hue},${Math.max(0,this.alpha)})`;
    ctx.fill();
  }

function loop(t){

    lastMouseBurst = t;

      const a=NODES[i], b=NODES[j];
      const dx=a.x-b.x, dy=a.y-b.y;
      const dist=Math.sqrt(dx*dx+dy*dy);
      if(dist < CONNECTION_DIST){
        const strength=(1-dist/CONNECTION_DIST);
        const opacity=strength*Math.min(a.alpha,b.alpha)*.7;
        if(opacity < .004) continue;
        ctx.beginPath();
        ctx.moveTo(a.x,a.y);
        // Gently curved filament — not a straight line
        const mx2=(a.x+b.x)/2 + (Math.random()-.5)*6;

        ctx.strokeStyle=`rgba(126,200,80,${opacity})`;
        ctx.lineWidth=strength*.7;
        ctx.stroke();
      }
    }
  }

  // Update and draw nodes
  for(let i=NODES.length-1;i>=0;i--){
    const dead = NODES[i].update(t);
    if(dead) NODES.splice(i,1);
    else NODES[i].draw();
  }

}

function openModal(name){
  const p=PRODUCTS[name];if(!p)return;
  // Ensure modal is a direct child of body so position:fixed works correctly
  // (forestSite is position:fixed + overflow:auto which traps fixed children)
  const overlay=document.getElementById('modalOverlay');
  if(overlay.parentElement!==document.body) document.body.appendChild(overlay);
  document.getElementById('mTag').textContent='Dr.Shrooms · '+(p.size||'');
  document.getElementById('mName').textContent=name;
  document.getElementById('mLatin').textContent=p.latin;
  document.getElementById('mOrigin').textContent='🌍 '+p.origin;
  document.getElementById('mTradition').textContent=p.tradition;
  const ul=document.getElementById('mBenefits');ul.innerHTML='';
  p.benefits.forEach(b=>{const li=document.createElement('li');li.textContent=b;ul.appendChild(li);});
  document.getElementById('mBestFor').textContent=p.bestFor;
  document.getElementById('mCompounds').textContent=p.activeCompounds;
  document.getElementById('mPrice').textContent=p.price;
  document.getElementById('mSize').textContent=p.size;
  const buyLink=document.getElementById('mBuyLink');
  if(buyLink){buyLink.href=p.shopUrl||'https://dr-shrooms-2.myshopify.com/';buyLink.textContent='Buy Online';}
  overlay.classList.add('open');
  document.body.style.overflow='hidden';
}
function closeModalDirect(){
  document.getElementById('modalOverlay').classList.remove('open');
  document.body.style.overflow='';
  // Restore forestSite scrollability if it was open
  const fs=document.getElementById('forestSite');
  if(fs&&fs.classList.contains('show')) fs.style.overflowY='auto';
}
function closeModal(e){if(e.target===document.getElementById('modalOverlay'))closeModalDirect();}
document.addEventListener('keydown',e=>{
  if(e.key==='Escape'){
    closeModalDirect();
    // Also close wholesale and land overlays
    const ws=document.getElementById('wholesaleSite');
    if(ws && ws.classList.contains('show')) closeWholesale();
    const lo=document.getElementById('landOverlay');
    if(lo && lo.classList.contains('show')) closeLandOverlay();
  }
});

(function initForestHeroCanvas(){
  const canvas = document.getElementById('forestHeroCanvas');
  if(!canvas) return;
  const ctx = canvas.getContext('2d');
  let W, H;
  function resize(){
    const hero = document.getElementById('forestHero');
    if(!hero) return;
    // offsetWidth/Height is 0 when visibility:hidden — fall back to window dimensions
    W = canvas.width = (hero.offsetWidth || window.innerWidth);
    H = canvas.height = (hero.offsetHeight || window.innerHeight);
  }

  const nodes = [];
  const NODE_COUNT = 80;

  class Node {
    constructor(){
      this.reset(true);
    }
    reset(init){
      this.x = Math.random() * (W||window.innerWidth);
      this.y = init ? Math.random() * (H||window.innerHeight) : (H||window.innerHeight) + 10;
      this.vx = (Math.random()-.5) * .15;
      this.vy = -(Math.random() * .18 + .03);
      this.size = Math.random() * 1.6 + .3;
      this.life = 0;
      this.maxLife = Math.random() * 600 + 300;
      this.alpha = 0;
      this.maxAlpha = Math.random() * .25 + .05;
      const r = Math.random();
      this.hue = r < .6 ? '126,200,80' : r < .85 ? '200,160,64' : '240,232,208';
    }
    update(){
      this.vx += (Math.random()-.5)*.006;
      this.vy += (Math.random()-.5)*.003 - .01;
      this.vx *= .99; this.vy *= .99;
      this.x += this.vx; this.y += this.vy;
      this.life++;
      const p = this.life / this.maxLife;
      if(p < .1) this.alpha = this.maxAlpha * (p/.1);
      else if(p > .85) this.alpha = this.maxAlpha * ((1-p)/.15);
      else this.alpha = this.maxAlpha;
      return this.life >= this.maxLife || this.y < -10;
    }
    draw(){
      ctx.beginPath();
      ctx.arc(this.x, this.y, this.size, 0, Math.PI*2);
      ctx.fillStyle = `rgba(${this.hue},${this.alpha})`;
      ctx.fill();
    }
  }

  function init(){
    resize();
    nodes.length = 0;
    for(let i=0;i<NODE_COUNT;i++) nodes.push(new Node());
  }

  let lastSpawn = 0;
  function loop(t){
    if(!canvas.parentElement) return;
    ctx.clearRect(0,0,W,H);

    if(t - lastSpawn > 800 && nodes.length < NODE_COUNT + 15){
      const n = new Node(); n.reset(false); nodes.push(n);
      lastSpawn = t;
    }

    // Draw connections
    for(let i=0;i<nodes.length;i++){
      for(let j=i+1;j<nodes.length;j++){
        const a=nodes[i], b=nodes[j];
        const dx=a.x-b.x, dy=a.y-b.y;
        const dist=Math.sqrt(dx*dx+dy*dy);
        if(dist < 150){
          const op=(1-dist/150)*Math.min(a.alpha,b.alpha)*.6;
          if(op < .003) continue;
          ctx.beginPath();
          ctx.moveTo(a.x,a.y);
          const mx=(a.x+b.x)/2+(Math.random()-.5)*5;
          const my=(a.y+b.y)/2+(Math.random()-.5)*5;
          ctx.quadraticCurveTo(mx,my,b.x,b.y);
          ctx.strokeStyle=`rgba(126,200,80,${op})`;
          ctx.lineWidth=(1-dist/150)*.6;
          ctx.stroke();
        }
      }
    }

    for(let i=nodes.length-1;i>=0;i--){
      if(nodes[i].update()){ nodes[i].reset(false); }
      else nodes[i].draw();
    }
    requestAnimationFrame(loop);
  }

  // Only init when forest site is shown
  const observer = new MutationObserver(()=>{
    const fs = document.getElementById('forestSite');
    if(fs && fs.classList.contains('show')){
      init();
      requestAnimationFrame(loop);
      observer.disconnect();
    }
  });
  const fs = document.getElementById('forestSite');
  if(fs) observer.observe(fs, {attributes:true, attributeFilter:['class']});

  window.addEventListener('resize', resize);
})();
